pragma solidity ^0.8.20;

import {ERC721} from "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";
import {Strings} from "@openzeppelin/contracts/utils/Strings.sol"; // 문자열 변환용

contract SpaceTaxi is ERC721, Ownable {
    using Strings for uint256;
    uint256 private _nextTokenId;

    constructor(address initialOwner)
        ERC721("SpaceTaxi", "ST")
        Ownable(initialOwner)
    {}

    // 1. 기본이 되는 IPFS 폴더 주소를 설정합니다.
    function _baseURI() internal pure override returns (string memory) {
        return "ipfs://bafybeiesiuvnolm2b5hz75glcixb3abxa7cqmmjb73dm5n74o7gek5ezze/";
    }

    // 2. 민팅 시 별도의 URI 설정 없이 ID만 올립니다.
    function safeMint(address to) public onlyOwner {
        uint256 tokenId = _nextTokenId++;
        _safeMint(to, tokenId);
    }

    // 3. ID 뒤에 .json을 붙여주는 핵심 함수입니다.
    function tokenURI(uint256 tokenId) public view override returns (string memory) {
        _requireOwned(tokenId); // 존재하는 토큰인지 확인

        string memory baseURI = _baseURI();
        return bytes(baseURI).length > 0 
            ? string(abi.encodePacked(baseURI, tokenId.toString(), ".json")) 
            : "";
    }
}
