pragma solidity 0.8.15;
 
contract SampleFallback {
    uint public lastValueSent;                //생성자 Contructor는 단한번만 호출됩니다.
    string public lastFunctionCalled;         
 
    receive() external payable {
        lastValueSent = msg.value;
        lastFunctionCalled = "receive";
    }
     fallback() external payable {   
        lastValueSent = msg.value;              //Fallback 함수는 단순 트랜잭션으로 이더를 받기 위해 존재
        lastFunctionCalled = "fallback";        //함수 호출 없이 단순히 컨트랙트 주소로 돈을 보낼때 작동하는 함수
    }                                           //이 함수는 가스소비 제한이 엄격하므로 복잡한 로직을 넣으면 안됨
                                                // Getter함수는 주로 view 함수입니다.
                                                //변수 값을 읽어오는 함수(Getter)는 블록체인의 상태를 변경하지 않기 때문에 view키워드를 사용
                                                //View/Pure 함수는 로컬 블록체인 노드에서만 실행
                                                //데이터를 읽기만 하거나(view), 아예 읽지도 않는 (pure)함수는 네트워크 전체이 전파될 필요가 없음
                                                //지불가능한 주소 (Payable addresses)가 있으며, 주소는 잔액(Balance)를 가집니다
                                                //이더를 받을수 있는 특정 타입(payable) 이 존재하며, 모든 주소는 가자의 잔액 정보를 가지고 있음  
                                             

}
