pragma solidity ^0.8.20; // 호환성을 위해 0.8.20 정도로 살짝 낮춤

import {ERC721} from "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import {ERC721URIStorage} from "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";

contract SpaceTaxi is ERC721, ERC721URIStorage, Ownable {
    uint256 private _nextTokenId;

    // initialOwner를 생성자에서 정확히 부모에게 전달
    constructor(address initialOwner)
        ERC721("SpaceTaxi", "ST")
        Ownable(initialOwner)
    {}

    function _baseURI() internal pure override returns (string memory) {
        return "ipfs://bafybeiarwls4masnmb3znjh5hkxj7ini7ttqjzedpmfcr3yx75rdufzaie/";
    }

    // URI를 직접 입력받는 대신, 베이스 주소를 활용하도록 수정
    function safeMint(address to) public onlyOwner {
        uint256 tokenId = _nextTokenId++;
        _safeMint(to, tokenId);
        // 만약 폴더 안에 0, 1, 2... 식의 JSON이 있다면 아래 줄은 필요 없습니다.
    }

    // 이하는 필수 오버라이드 함수들
    function tokenURI(uint256 tokenId) public view override(ERC721, ERC721URIStorage) returns (string memory) {
        return super.tokenURI(tokenId);
    }

    function supportsInterface(bytes4 interfaceId) public view override(ERC721, ERC721URIStorage) returns (bool) {
        return super.supportsInterface(interfaceId);
    }
}
